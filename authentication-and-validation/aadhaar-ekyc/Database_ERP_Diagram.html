Database Schema & ERP Diagram

Firestore Collections, Field Types, Entity Relationships

1. Firestore Collections Schema

📁 artists/{artistId}
│
├── personalInfo
│   ├── fullName: String          // From registration form
│   ├── email: String
│   ├── phone: String
│   ├── city: String
│   └── category: String          // Singer, DJ, Dancer, etc.
│
├── 🔐 kycVerification
│   ├── method: String            // "aadhaar_otp" | "aadhaar_xml"
│   ├── isVerified: Boolean       // true / false
│   ├── verifiedAt: Timestamp    // When verification completed
│   ├── verifiedName: String     // Name extracted from Aadhaar
│   ├── dateOfBirth: String      // "YYYY-MM-DD" from Aadhaar
│   ├── ageAtVerification: Number // Calculated age at time of KYC
│   ├── ageTier: String          // "adult" (18+) | "minor" (<18)
│   ├── gender: String            // M / F / T from Aadhaar
│   ├── nameMatchScore: Number   // % match: form name vs Aadhaar
│   ├── verificationAttempts: Number
│   ├── lastAttemptAt: Timestamp
│   └── ⚠️ aadhaarNumber: NULL    // NEVER STORED — privacy
│
├── profileStatus
│   ├── status: String            // "pending_kyc" | "pending_review" |
│   │                              // "approved" | "rejected" | "re_verify"
│   ├── reviewedBy: String       // Admin UID who reviewed
│   ├── reviewedAt: Timestamp
│   ├── rejectionReason: String  // If rejected
│   └── isLive: Boolean          // Profile visible to fans?
│
├── artistProfile
│   ├── stageName: String
│   ├── bio: String
│   ├── genres: Array<String>
│   ├── socialLinks: Map
│   ├── profilePhoto: String     // Firebase Storage URL
│   └── portfolioVideo: String  // YouTube/link
│
└── consent
    ├── termsAccepted: Boolean
    ├── termsAcceptedAt: Timestamp
    ├── copyrightDeclared: Boolean
    ├── iprsAcknowledged: Boolean
    └── dataConsentGiven: Boolean


📁 kycAuditLog/{logId}
├── artistId: String             // FK → artists/{artistId}
├── action: String               // "otp_sent" | "otp_verified" | "xml_uploaded" |
│                                 // "age_check_passed" | "age_check_failed" |
│                                 // "admin_approved" | "admin_rejected" | "content_gated"
├── timestamp: Timestamp
├── performedBy: String          // "system" | admin UID
├── metadata: Map               // Additional context (error codes, etc.)
└── ipAddress: String            // For security audit


📁 adminActions/{actionId}
├── adminId: String              // FK → admins
├── artistId: String             // FK → artists
├── action: String               // "approve" | "reject" | "request_reverify"
├── reason: String
├── timestamp: Timestamp
└── previousStatus: String


📁 users/{userId}                 // For under-18 users in restricted mode
├── ageTier: String              // "adult" | "under_18"
├── dateOfBirth: String         // Stored for auto-upgrade on birthday
├── contentFilter: String       // "all" | "age_appropriate"
├── kycVerifiedAt: Timestamp
└── nextAgeCheckDate: String    // Their 18th birthday — triggers upgrade
⚠️ Critical Privacy Rule: The 12-digit Aadhaar number is NEVER stored anywhere in the database. Only: verification status (boolean), verified name, DOB, gender, and masked reference ID (last 4 digits) are persisted.

3. Firestore Security Rules (Key Rules)

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Artists can read their own KYC data, never write directly
    match /artists/{artistId} {
      allow read: if request.auth.uid == artistId;
      allow write: if false;  // Only Cloud Functions can write KYC data
    }

    // Audit logs — read only for admins
    match /kycAuditLog/{logId} {
      allow read: if isAdmin(request.auth.uid);
      allow write: if false;  // Only Cloud Functions
    }

    // Admin actions — admins can read, Cloud Functions write
    match /adminActions/{actionId} {
      allow read: if isAdmin(request.auth.uid);
      allow write: if false;
    }

    function isAdmin(uid) {
      return exists(/databases/$(database)/documents/admins/$(uid));
    }
  }
}
